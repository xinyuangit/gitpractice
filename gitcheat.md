#Git Cheatsheet

git config --global user.name "[name]
git config --global user.email "[email address]
git config --list
git init
git remote add origin [URL of your remote repo]
git remote -v
git status
git log
git add .
git commit -m "[clear message describing the changes you made]"
git push origin master
